def dig?(index, *keys) #

Retrieves the value object corresponding to the each key objects repeatedly. The type needs to implement #[]? method to works.

h = { foo: {bar: {baz: 5}}}

h.dig?(:foo, :bar, :baz)           #=> 5
h.dig?(:foo, :blah)                #=> nil
